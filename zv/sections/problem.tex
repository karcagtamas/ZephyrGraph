\begin{frame}{Problem Statement}
    \begin{alertblock}{Challenges in Testing Complex Systems}
        \begin{itemize}[<+->]\itemsep9pt
            \item Ensuring business logic accuracy is difficult in modern, interconnected systems.
            \item Traditional manual testing struggles with scalability and completeness.
        \end{itemize}
    \end{alertblock}

    \begin{alertblock}{Cause-Effect Graphs}
        \begin{itemize}[<+->]\itemsep9pt
            \item \textbf{Strength}: 
                \begin{itemize}
                    \item Provide a structured and visual representation of relationships between causes (inputs) and effects (outputs).
                    \item Aid in understanding and modeling complex dependencies.
                \end{itemize}
            \item \textbf{Challenge}:
                \begin{itemize}
                    \item Transitioning from visual cause-effect graphs to actionable test cases is cumbersome.
                    \item Current methods rely on manual processes or hard-to-scale specialized solutions.
                \end{itemize}
        \end{itemize}
    \end{alertblock}
\end{frame}
