\begin{frame}{Introduction}
    \begin{alertblock}{Background Context}
        \begin{itemize}[<+->]\itemsep9pt
            \item \textbf{Problem Domain}: Modern software testing often requires efficient tools to define, analyze, and manage complex business rules and relationships.
            \item \textbf{Motivation}: Current solutions for cause-effect graph generation and testing are either cumbersome or lack reusability, making the testing process inefficient.
        \end{itemize}
    \end{alertblock}
    \begin{alertblock}{Objectives}
        \begin{itemize}[<+->]\itemsep9pt
            \item Develop a tool to define, transform, and visualize cause-effect graphs.
            \item Enable reusable, scalable, and efficient solutions for test case generation.
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{Introduction}
    \begin{alertblock}{Scope}
        \begin{itemize}[<+->]\itemsep9pt
            \item Focuses on creating a custom graph language for modeling cause-effect structure and relationships.
            \item Provides transformations and export capabilities for effective testing, including Boundary Value Analysis (BVA) and General Predicate Testing (GPT).
        \end{itemize}
    \end{alertblock}
\end{frame}
