\begin{frame}{Introduction}
    \begin{alertblock}{Background and Motivation}
        \begin{itemize}[<+->]\itemsep9pt
            \item \textbf{Evolving Digital Landscape}: Increased focus on security and reliability to protect sensitive data.
            \item \textbf{Challenges in Software Testing}
                \begin{itemize}
                    \item Ensuring the correctness of complex business logic is critical.
                    \item Traditional manual testing approaches struggle with coverage and scalability.
                    \item Incomplete tests can lead to undetected defects in critical applications.
                \end{itemize}
            \item \textbf{Cause-Effect Graphs in Testing}
                \begin{itemize}
                    \item Provide a structured way to model business logic and dependencies.
                    \item Visualize relationships between causes (inputs) and effects (outputs) effectively.
                    \item Current methods for generating test cases are manual, inefficient, and lack scalability.
                \end{itemize}
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{Introduction}
    \begin{alertblock}{Objectives}
        \begin{itemize}[<+->]\itemsep9pt
            \item Develop an automated framework for transforming textual cause-effect graphs into:
                \begin{itemize}
                    \item Logical formulas.
                    \item Decision table.
                    \item Optimized test sets.
                \end{itemize}
            \item Improve test coverage and minimize manual effort in validating business-critical applications.
        \end{itemize}
    \end{alertblock}
\end{frame}
